1. IAA LTV + eCPM + Freq 分析 (全产品通用)

-- CTE 结构：
-- 1. new_users: 筛选新注册用户 (been_reg_days=1)
-- 2. daily_ad_metrics: 关联活跃表，计算每日广告播放和收入
-- 3. lifecycle_metrics: 按 LT 窗口聚合
-- 4. 最终 SELECT: CUBE(platform) 自动生成分平台 + ALL

-- 核心指标：
-- Freq/DAU = SUM(adplay) / SUM(active_days)
-- eCPM = SUM(revenue) * 1000 / SUM(adplay)
-- IAA LTV = SUM(revenue) / COUNT(DISTINCT user_id)

2. 3PP 支付转化漏斗

-- 各步打点：
-- Step1 访问: open_recharge (uuid排重)
-- Step2 登录: avatar_suc (排重)
-- Step3 点击商品: create_order (排重)
-- Step4 选择支付: select_payment (排重)
-- Step5 渠道下单: server_web_order_suc (排重)
-- Step6 支付成功: omni_server_payment
--   WHERE pay_channel_id = 11 AND payment > 0
--   AND status IN (1, 3) AND order_from != 'webPay'

3. UNO2 广告数据 (FULL OUTER JOIN)

-- 播放次数: dw_ods_mn08.c_client_app_ad_log WHERE log_subtype='adplay' AND status=1
-- 广告收入: dw_ods_mn08.c_client_app_ad_log WHERE log_subtype='advalue'
-- 安卓需 /1000000: CASE WHEN platform='IOS' THEN value ELSE value/1000000 END
-- 合并: FULL OUTER JOIN ON date + role_id + platform

4. 三方支付：内嵌收入 / 主站收入拆分

-- 数据源: dw_ods_sdk.omni_server_payment
-- game_id: UNO=1000000, P10=1019, SKB=1000008, UNO2=1000014
SELECT
    COUNT(DISTINCT account_id)                      AS pay_user,
    SUM(payment)                                    AS third_pay_revenue,
    SUM(IF(sub_order_from = '3ppingame', payment, 0)) AS third_pay_ingame,  -- 内嵌收入
    SUM(IF(sub_order_from IS NULL, payment, 0))       AS third_pay_web      -- 主站收入
FROM dw_ods_sdk.omni_server_payment
WHERE date = '2026-01-01'
  AND pay_channel_id = 11
  AND payment > 0
  AND order_from = 'storePay'
  AND status IN (1, 3, 4)
  AND game_id = 1000000  -- 按产品替换 game_id

5. 三方支付：二选一弹窗曝光/选择统计 (UNO)

-- 数据源: dw_ods_mn01.c_client_app_payment_selectchannel
SELECT
    date,
    CAST(account_id AS VARCHAR) AS account_id,
    COUNT(IF(extra_col_action = 'openUI', 1, NULL))    AS open_cnt,       -- 打开弹窗
    COUNT(IF(extra_col_action = 'close', 1, NULL))     AS close_cnt,      -- 关闭弹窗
    COUNT(IF(extra_col_action = 'selectIAP', 1, NULL)) AS select_iap_cnt, -- 选择内购
    COUNT(IF(extra_col_action = 'selectweb', 1, NULL)) AS select_web_cnt  -- 选择三方
FROM dw_ods_mn01.c_client_app_payment_selectchannel
WHERE date BETWEEN '2026-01-01' AND '2026-01-07'
GROUP BY 1, 2

6. UNO role_id → account_id 映射

-- UNO 客户端用 role_id，SDK 用 account_id
-- 通过 c_app_ad_log 表关联获取映射
SELECT DISTINCT
    roleid AS role_id,
    account_id
FROM dw_ods_mn01.c_app_ad_log
WHERE date = '2026-01-01'
  AND roleid = '目标role_id'
